import time
import board
import busio
import adafruit_ads1x15.ads1115 as ADS
from adafruit_ads1x15.analog_in import AnalogIn

# –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è I2C –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å–∞
i2c = busio.I2C(board.SCL, board.SDA)

# –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è ADS1115
ads = ADS.ADS1115(i2c)

# –ù–∞—Å—Ç—Ä–æ–π–∫–∞ –∫–∞–Ω–∞–ª–∞ A0 –¥–ª—è –ø–æ—Ç–µ–Ω—Ü–∏–æ–º–µ—Ç—Ä–∞
potentiometer = AnalogIn(ads, ADS.P0)

# –£—Å—Ç–∞–Ω–æ–≤–∫–∞ –¥–∏–∞–ø–∞–∑–æ–Ω–∞ –∏–∑–º–µ—Ä–µ–Ω–∏–π (¬±4.096V)
ads.gain = 1

# –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏—è –∑–Ω–∞—á–µ–Ω–∏—è –≤ –ø—Ä–æ—Ü–µ–Ω—Ç—ã
def map_to_percent(value, in_min=0, in_max=26000):
    return int((value - in_min) * 100 / (in_max - in_min))

# –§—É–Ω–∫—Ü–∏—è –¥–ª—è —Å–æ–∑–¥–∞–Ω–∏—è ASCII-–≥—Ä–∞—Ñ–∏–∫–∞
def create_ascii_bar(percent, length=50):
    filled_length = int(length * percent / 100)
    bar = '‚ñà' * filled_length + '‚ñë' * (length - filled_length)
    return f"[{bar}] {percent}%"

# –û—Å–Ω–æ–≤–Ω–æ–π —Ü–∏–∫–ª
try:
    print("–ú–æ–¥—É–ª—å –ê–¶–ü ADS1115 –≥–æ—Ç–æ–≤ –∫ —Ä–∞–±–æ—Ç–µ!")
    print("–í—Ä–∞—â–∞–π—Ç–µ –ø–æ—Ç–µ–Ω—Ü–∏–æ–º–µ—Ç—Ä –¥–ª—è –∏–∑–º–µ–Ω–µ–Ω–∏—è –∑–Ω–∞—á–µ–Ω–∏–π")
    print("–ù–∞–∂–º–∏—Ç–µ Ctrl+C –¥–ª—è –≤—ã—Ö–æ–¥–∞")
    print()

    # –ù–µ–±–æ–ª—å—à–∞—è –∑–∞–¥–µ—Ä–∂–∫–∞ –¥–ª—è –∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏–∏
    time.sleep(0.5)

    while True:
        # –°—á–∏—Ç—ã–≤–∞–µ–º –∑–Ω–∞—á–µ–Ω–∏–µ —Å –ø–æ—Ç–µ–Ω—Ü–∏–æ–º–µ—Ç—Ä–∞
        raw_value = potentiometer.value
        voltage = potentiometer.voltage

        # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º –≤ –ø—Ä–æ—Ü–µ–Ω—Ç—ã
        percent = map_to_percent(raw_value)

        # –°–æ–∑–¥–∞–µ–º –≤–∏–∑—É–∞–ª—å–Ω–æ–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏–µ
        bar = create_ascii_bar(percent)

        # –û—á–∏—â–∞–µ–º –ø—Ä–µ–¥—ã–¥—É—â—É—é —Å—Ç—Ä–æ–∫—É –∏ –≤—ã–≤–æ–¥–∏–º –Ω–æ–≤—É—é –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—é

        # –ó–∞–¥–µ—Ä–∂–∫–∞ –¥–ª—è —Å—Ç–∞–±–∏–ª–∏–∑–∞—Ü–∏–∏ –ø–æ–∫–∞–∑–∞–Ω–∏–π
        time.sleep(0.1)
