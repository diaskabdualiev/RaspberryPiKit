**–§–∞–π–ª app.py**

.. code-block:: python

   from flask import Flask, render_template, jsonify
   import time
   import board
   import digitalio
   import threading

   app = Flask(__name__)

   # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –≤—ã–≤–æ–¥–æ–≤ —ç–Ω–∫–æ–¥–µ—Ä–∞
   # –í—ã—Ö–æ–¥—ã A –∏ B —ç–Ω–∫–æ–¥–µ—Ä–∞ –ø–æ–¥–∫–ª—é—á–µ–Ω—ã –∫ GPIO17 –∏ GPIO18 —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–µ–Ω–Ω–æ
   pin_a = digitalio.DigitalInOut(board.D17)
   pin_b = digitalio.DigitalInOut(board.D18)
   pin_a.direction = digitalio.Direction.INPUT
   pin_b.direction = digitalio.Direction.INPUT
   pin_a.pull = digitalio.Pull.UP  # –ü–æ–¥—Ç—è–∂–∫–∞ –∫ –ø–∏—Ç–∞–Ω–∏—é
   pin_b.pull = digitalio.Pull.UP  # –ü–æ–¥—Ç—è–∂–∫–∞ –∫ –ø–∏—Ç–∞–Ω–∏—é

   # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –∫–Ω–æ–ø–∫–∏ —ç–Ω–∫–æ–¥–µ—Ä–∞
   button = digitalio.DigitalInOut(board.D27)  # –ö–Ω–æ–ø–∫–∞ –Ω–∞ GPIO27
   button.direction = digitalio.Direction.INPUT
   button.pull = digitalio.Pull.UP  # –ü–æ–¥—Ç—è–∂–∫–∞ –∫ VCC (–∫–Ω–æ–ø–∫–∞ –∑–∞–º—ã–∫–∞–µ—Ç –Ω–∞ GND)

   # –ì–ª–æ–±–∞–ª—å–Ω—ã–µ –ø–µ—Ä–µ–º–µ–Ω–Ω—ã–µ –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è —Å–æ—Å—Ç–æ—è–Ω–∏—è
   counter = 0
   button_state = False
   last_button_state = False
   last_a_state = pin_a.value
   last_direction = ""
   events = []  # –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è –∏—Å—Ç–æ—Ä–∏–∏ —Å–æ–±—ã—Ç–∏–π

   # –ë–ª–æ–∫–∏—Ä–æ–≤–∫–∞ –¥–ª—è –º–Ω–æ–≥–æ–ø–æ—Ç–æ—á–Ω–æ–≥–æ –¥–æ—Å—Ç—É–ø–∞
   lock = threading.Lock()

   # –§—É–Ω–∫—Ü–∏—è –¥–ª—è –æ–ø—Ä–æ—Å–∞ —ç–Ω–∫–æ–¥–µ—Ä–∞ –≤ –æ—Ç–¥–µ–ª—å–Ω–æ–º –ø–æ—Ç–æ–∫–µ
   def encoder_polling():
       global counter, button_state, last_button_state, last_a_state, last_direction, events
       
       try:
           print("–†–æ—Ç–æ—Ä–Ω—ã–π —ç–Ω–∫–æ–¥–µ—Ä: –ø–æ–≤–æ—Ä–∞—á–∏–≤–∞–π—Ç–µ —Ä—É—á–∫—É –∏–ª–∏ –Ω–∞–∂–º–∏—Ç–µ –Ω–∞ –Ω–µ–µ")
           
           while True:
               with lock:
                   # –°—á–∏—Ç—ã–≤–∞–µ–º —Ç–µ–∫—É—â–µ–µ —Å–æ—Å—Ç–æ—è–Ω–∏–µ –≤—ã–≤–æ–¥–æ–≤ —ç–Ω–∫–æ–¥–µ—Ä–∞
                   a_state = pin_a.value
                   b_state = pin_b.value
                   
                   # –ï—Å–ª–∏ —Å–æ—Å—Ç–æ—è–Ω–∏–µ –≤—ã–≤–æ–¥–∞ A –∏–∑–º–µ–Ω–∏–ª–æ—Å—å, –∑–Ω–∞—á–∏—Ç –ø—Ä–æ–∏–∑–æ—à–ª–æ –≤—Ä–∞—â–µ–Ω–∏–µ
                   if a_state != last_a_state:
                       # –û–ø—Ä–µ–¥–µ–ª—è–µ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–µ –≤—Ä–∞—â–µ–Ω–∏—è —Å—Ä–∞–≤–Ω–∏–≤–∞—è —Å–æ—Å—Ç–æ—è–Ω–∏—è –≤—ã–≤–æ–¥–æ–≤ A –∏ B
                       if b_state != a_state:
                           direction = "–ø–æ —á–∞—Å–æ–≤–æ–π —Å—Ç—Ä–µ–ª–∫–µ"
                           counter += 1
                       else:
                           direction = "–ø—Ä–æ—Ç–∏–≤ —á–∞—Å–æ–≤–æ–π —Å—Ç—Ä–µ–ª–∫–∏"
                           counter -= 1
                       
                       # –°–æ—Ö—Ä–∞–Ω—è–µ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–µ –∏ –¥–æ–±–∞–≤–ª—è–µ–º —Å–æ–±—ã—Ç–∏–µ
                       last_direction = direction
                       events.append(f"–í—Ä–∞—â–µ–Ω–∏–µ {direction}, –°—á–µ—Ç—á–∏–∫: {counter}")
                       # –û–≥—Ä–∞–Ω–∏—á–∏–≤–∞–µ–º –∏—Å—Ç–æ—Ä–∏—é —Å–æ–±—ã—Ç–∏–π –¥–æ 10
                       if len(events) > 10:
                           events = events[-10:]
                   
                   # –û–±–Ω–æ–≤–ª—è–µ–º –ø–æ—Å–ª–µ–¥–Ω–µ–µ —Å–æ—Å—Ç–æ—è–Ω–∏–µ –≤—ã–≤–æ–¥–∞ A
                   last_a_state = a_state
                   
                   # –û–±—Ä–∞–±–æ—Ç–∫–∞ –Ω–∞–∂–∞—Ç–∏—è –∫–Ω–æ–ø–∫–∏
                   button_state = not button.value  # –ò–Ω–≤–µ—Ä—Ç–∏—Ä—É–µ–º –∑–Ω–∞—á–µ–Ω–∏–µ
                   
                   # –ü—Ä–æ–≤–µ—Ä—è–µ–º –∏–∑–º–µ–Ω–µ–Ω–∏–µ —Å–æ—Å—Ç–æ—è–Ω–∏—è –∫–Ω–æ–ø–∫–∏ (–æ–±–Ω–∞—Ä—É–∂–µ–Ω–∏–µ —Ñ—Ä–æ–Ω—Ç–∞)
                   if button_state and not last_button_state:
                       events.append(f"–ö–Ω–æ–ø–∫–∞ –Ω–∞–∂–∞—Ç–∞! –°–±—Ä–æ—Å —Å—á–µ—Ç—á–∏–∫–∞ —Å {counter} –Ω–∞ 0")
                       counter = 0
                   
                   # –û–±–Ω–æ–≤–ª—è–µ–º –ø–æ—Å–ª–µ–¥–Ω–µ–µ —Å–æ—Å—Ç–æ—è–Ω–∏–µ –∫–Ω–æ–ø–∫–∏
                   last_button_state = button_state
               
               # –ù–µ–±–æ–ª—å—à–∞—è –∑–∞–¥–µ—Ä–∂–∫–∞ –¥–ª—è —Å—Ç–∞–±–∏–ª–∏–∑–∞—Ü–∏–∏
               time.sleep(0.01)
               
       except Exception as e:
           print(f"–û—à–∏–±–∫–∞ –≤ –ø–æ—Ç–æ–∫–µ –æ–ø—Ä–æ—Å–∞ —ç–Ω–∫–æ–¥–µ—Ä–∞: {e}")

   # –ú–∞—Ä—à—Ä—É—Ç –¥–ª—è –≥–ª–∞–≤–Ω–æ–π —Å—Ç—Ä–∞–Ω–∏—Ü—ã
   @app.route('/')
   def index():
       return render_template('index.html')

   # API –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è —Ç–µ–∫—É—â–µ–≥–æ —Å–æ—Å—Ç–æ—è–Ω–∏—è —ç–Ω–∫–æ–¥–µ—Ä–∞
   @app.route('/api/encoder-state')
   def encoder_state():
       with lock:
           return jsonify({
               'counter': counter,
               'button_state': button_state,
               'last_direction': last_direction,
               'events': events
           })

   # –ó–∞–ø—É—Å–∫ –ø–æ—Ç–æ–∫–∞ –æ–ø—Ä–æ—Å–∞ —ç–Ω–∫–æ–¥–µ—Ä–∞
   def start_encoder_thread():
       encoder_thread = threading.Thread(target=encoder_polling, daemon=True)
       encoder_thread.start()

   if __name__ == '__main__':
       # –ó–∞–ø—É—Å–∫–∞–µ–º –ø–æ—Ç–æ–∫ –¥–ª—è –æ–ø—Ä–æ—Å–∞ —ç–Ω–∫–æ–¥–µ—Ä–∞
       start_encoder_thread()
       
       # –ó–∞–ø—É—Å–∫–∞–µ–º –≤–µ–±-—Å–µ—Ä–≤–µ—Ä Flask
       app.run(host='0.0.0.0', port=5000, debug=False, threaded=True)
