   import pwmio
   import time

   # –ù–∞—Å—Ç—Ä–æ–π–∫–∞ –ø–∏–Ω–∞ –¥–ª—è –ø–∞—Å—Å–∏–≤–Ω–æ–≥–æ –∑—É–º–º–µ—Ä–∞ —Å –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ–º PWM

   # –°–ª–æ–≤–∞—Ä—å –Ω–æ—Ç –∏ –∏—Ö —á–∞—Å—Ç–æ—Ç –≤ –ì—Ü
   NOTES = {
       'C4': 262,  # –î–æ —á–µ—Ç–≤–µ—Ä—Ç–æ–π –æ–∫—Ç–∞–≤—ã
       'D4': 294,  # –†–µ
       'E4': 330,  # –ú–∏
       'F4': 349,  # –§–∞
       'G4': 392,  # –°–æ–ª—å
       'A4': 440,  # –õ—è
       'B4': 494,  # –°–∏
       'C5': 523,  # –î–æ –ø—è—Ç–æ–π –æ–∫—Ç–∞–≤—ã
       'REST': 0,  # –ü–∞—É–∑–∞
   }

   # –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø—Ä–æ–∏–≥—Ä—ã–≤–∞–Ω–∏—è –Ω–æ—Ç—ã –∑–∞–¥–∞–Ω–Ω–æ–π –¥–ª–∏—Ç–µ–ª—å–Ω–æ—Å—Ç–∏
   def play_note(note, duration):
       if note == 'REST':
           # –ü–∞—É–∑–∞ - –ø—Ä–æ—Å—Ç–æ –∂–¥–µ–º
           buzzer.duty_cycle = 0
           time.sleep(duration)
       else:

           # –£—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º –≥—Ä–æ–º–∫–æ—Å—Ç—å (50% –æ—Ç –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–π)

           time.sleep(duration)
           # –í—ã–∫–ª—é—á–∞–µ–º –∑–≤—É–∫
           buzzer.duty_cycle = 0
           # –ù–µ–±–æ–ª—å—à–∞—è –ø–∞—É–∑–∞ –º–µ–∂–¥—É –Ω–æ—Ç–∞–º–∏
           time.sleep(0.05)

   # –ü—Ä–æ—Å—Ç–∞—è –º–µ–ª–æ–¥–∏—è - "–ú–∞–ª–µ–Ω—å–∫–∞—è –∑–≤–µ–∑–¥–æ—á–∫–∞" (Twinkle Twinkle Little Star)

       ('C4', 0.3), ('C4', 0.3), ('G4', 0.3), ('G4', 0.3),
       ('A4', 0.3), ('A4', 0.3), ('G4', 0.6),
       ('F4', 0.3), ('F4', 0.3), ('E4', 0.3), ('E4', 0.3),
       ('D4', 0.3), ('D4', 0.3), ('C4', 0.6),
   ]

   try:
       print("–í–æ—Å–ø—Ä–æ–∏–∑–≤–µ–¥–µ–Ω–∏–µ –º–µ–ª–æ–¥–∏–∏ –Ω–∞ –ø–∞—Å—Å–∏–≤–Ω–æ–º –∑—É–º–º–µ—Ä–µ")

           # –ü–∞—É–∑–∞ –ø–µ—Ä–µ–¥ –ø–æ–≤—Ç–æ—Ä–µ–Ω–∏–µ–º
           time.sleep(1)
               
   except KeyboardInterrupt:
       # –í—ã–∫–ª—é—á–∞–µ–º –∑—É–º–º–µ—Ä –ø—Ä–∏ –∑–∞–≤–µ—Ä—à–µ–Ω–∏–∏

       print("\n–ü—Ä–æ–≥—Ä–∞–º–º–∞ –∑–∞–≤–µ—Ä—à–µ–Ω–∞")
