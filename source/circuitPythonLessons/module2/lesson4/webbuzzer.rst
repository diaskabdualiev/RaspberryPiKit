   # –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø–æ–¥–∞—á–∏ –∑–≤—É–∫–æ–≤–æ–≥–æ —Å–∏–≥–Ω–∞–ª–∞

       status = "–û–∂–∏–¥–∞–Ω–∏–µ"
       buzzer_active = False

   # –§—É–Ω–∫—Ü–∏—è –¥–ª—è –∑–∞–ø—É—Å–∫–∞ –¥–≤—É—Ö —Å–∏–≥–Ω–∞–ª–æ–≤
   def double_beep():
       global status, buzzer_active
       
       if buzzer_active:
           return
       
       buzzer_active = True
       
       # –ü–µ—Ä–≤—ã–π —Å–∏–≥–Ω–∞–ª

       status = "–ó—É–º–º–µ—Ä –í–ö–õ (1/2)"

       time.sleep(0.2)
       
       # –í—Ç–æ—Ä–æ–π —Å–∏–≥–Ω–∞–ª

       status = "–ó—É–º–º–µ—Ä –í–ö–õ (2/2)"

       buzzer_pin.value = False
       
       status = "–û–∂–∏–¥–∞–Ω–∏–µ"

   @app.route('/')
   def index():

   @app.route('/beep/<beep_type>')
   def trigger_beep(beep_type):

           thread = threading.Thread(target=beep, args=(0.2,))
       elif beep_type == 'double':

       elif beep_type == 'long':

           thread.daemon = True
           thread.start()
       
       return redirect(url_for('index'))

       # –†–µ–≥–∏—Å—Ç—Ä–∏—Ä—É–µ–º —Ñ—É–Ω–∫—Ü–∏—é –æ—á–∏—Å—Ç–∫–∏
