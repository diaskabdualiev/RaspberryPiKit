**–§–∞–π–ª app.py**

.. code-block:: python

   from flask import Flask, render_template, jsonify
   import time
   import board
   import busio
   import adafruit_ads1x15.ads1115 as ADS
   from adafruit_ads1x15.analog_in import AnalogIn
   import threading

   app = Flask(__name__)

   # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è I2C –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å–∞
   i2c = busio.I2C(board.SCL, board.SDA)

   # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è ADS1115
   ads = ADS.ADS1115(i2c)

   # –ù–∞—Å—Ç—Ä–æ–π–∫–∞ –∫–∞–Ω–∞–ª–∞ A0 –¥–ª—è –ø–æ—Ç–µ–Ω—Ü–∏–æ–º–µ—Ç—Ä–∞
   potentiometer = AnalogIn(ads, ADS.P0)

   # –£—Å—Ç–∞–Ω–æ–≤–∫–∞ –¥–∏–∞–ø–∞–∑–æ–Ω–∞ –∏–∑–º–µ—Ä–µ–Ω–∏–π (¬±4.096V)
   ads.gain = 1

   # –ì–ª–æ–±–∞–ª—å–Ω—ã–µ –ø–µ—Ä–µ–º–µ–Ω–Ω—ã–µ –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è —Ç–µ–∫—É—â–∏—Ö –∑–Ω–∞—á–µ–Ω–∏–π
   current_value = 0
   current_voltage = 0
   current_percent = 0

   # –ë–ª–æ–∫–∏—Ä–æ–≤–∫–∞ –¥–ª—è –º–Ω–æ–≥–æ–ø–æ—Ç–æ—á–Ω–æ–≥–æ –¥–æ—Å—Ç—É–ø–∞
   lock = threading.Lock()

   # –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏—è –∑–Ω–∞—á–µ–Ω–∏—è –≤ –ø—Ä–æ—Ü–µ–Ω—Ç—ã
   def map_to_percent(value, in_min=0, in_max=26000):
       return int((value - in_min) * 100 / (in_max - in_min))

   # –§—É–Ω–∫—Ü–∏—è –æ–ø—Ä–æ—Å–∞ –ê–¶–ü –≤ –æ—Ç–¥–µ–ª—å–Ω–æ–º –ø–æ—Ç–æ–∫–µ
   def adc_polling():
       global current_value, current_voltage, current_percent
       
       try:
           while True:
               # –°—á–∏—Ç—ã–≤–∞–µ–º –∑–Ω–∞—á–µ–Ω–∏–µ —Å –ø–æ—Ç–µ–Ω—Ü–∏–æ–º–µ—Ç—Ä–∞
               raw_value = potentiometer.value
               voltage = potentiometer.voltage
               
               # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º –≤ –ø—Ä–æ—Ü–µ–Ω—Ç—ã
               percent = map_to_percent(raw_value)
               
               # –û–±–Ω–æ–≤–ª—è–µ–º –≥–ª–æ–±–∞–ª—å–Ω—ã–µ –ø–µ—Ä–µ–º–µ–Ω–Ω—ã–µ
               with lock:
                   current_value = raw_value
                   current_voltage = voltage
                   current_percent = percent
               
               # –ó–∞–¥–µ—Ä–∂–∫–∞ –¥–ª—è —Å—Ç–∞–±–∏–ª–∏–∑–∞—Ü–∏–∏ –ø–æ–∫–∞–∑–∞–Ω–∏–π
               time.sleep(0.1)
               
       except Exception as e:
           print(f"–û—à–∏–±–∫–∞ –≤ –ø–æ—Ç–æ–∫–µ –æ–ø—Ä–æ—Å–∞ –ê–¶–ü: {e}")

   # –ú–∞—Ä—à—Ä—É—Ç –¥–ª—è –≥–ª–∞–≤–Ω–æ–π —Å—Ç—Ä–∞–Ω–∏—Ü—ã
   @app.route('/')
   def index():
       return render_template('index.html')

   # API –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è —Ç–µ–∫—É—â–∏—Ö –∑–Ω–∞—á–µ–Ω–∏–π
   @app.route('/api/adc-state')
   def adc_state():
       with lock:
           return jsonify({
               'raw_value': current_value,
               'voltage': current_voltage,
               'percent': current_percent
           })

   # –ó–∞–ø—É—Å–∫ –ø–æ—Ç–æ–∫–∞ –æ–ø—Ä–æ—Å–∞ –ê–¶–ü
   def start_adc_thread():
       adc_thread = threading.Thread(target=adc_polling, daemon=True)
       adc_thread.start()

   if __name__ == '__main__':
       # –ó–∞–ø—É—Å–∫–∞–µ–º –ø–æ—Ç–æ–∫ –¥–ª—è –æ–ø—Ä–æ—Å–∞ –ê–¶–ü
       start_adc_thread()
       
       # –ó–∞–ø—É—Å–∫–∞–µ–º –≤–µ–±-—Å–µ—Ä–≤–µ—Ä Flask
       app.run(host='0.0.0.0', port=5000, debug=False, threaded=True)
