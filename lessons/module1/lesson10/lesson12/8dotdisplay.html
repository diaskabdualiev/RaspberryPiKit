        with canvas(device) as draw:
            text(draw, (-i, 0), message, fill="white", font=proportional(LCD_FONT))
        sleep(delay)

# –§—É–Ω–∫—Ü–∏—è –¥–ª—è –≤—ã–≤–æ–¥–∞ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è —á–µ—Ä–µ–∑ –º–∞—Å—Å–∏–≤ –ø–∏–∫—Å–µ–ª–µ–π
def display_array(matrix):
    """ matrix - –¥–≤—É–º–µ—Ä–Ω—ã–π –º–∞—Å—Å–∏–≤ 8x16 (–¥–ª—è –¥–≤—É—Ö –¥–∏—Å–ø–ª–µ–µ–≤) """
    with canvas(device) as draw:
        for y, row in enumerate(matrix):
            for x, pixel in enumerate(row):
                if pixel:  # –ï—Å–ª–∏ 1, —Ç–æ —Ä–∏—Å—É–µ–º —Ç–æ—á–∫—É
                    draw.point((x, y), fill="white")

# –ü—Ä–∏–º–µ—Ä –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è: –ø—Ä—è–º–æ—É–≥–æ–ª—å–Ω–∏–∫

# –û—Å–Ω–æ–≤–Ω–æ–π —Ü–∏–∫–ª
try:
    display_text("Hi!")  # –û—Ç–æ–±—Ä–∞–∑–∏—Ç—å "Hi!" –Ω–∞ –¥–≤—É—Ö –¥–∏—Å–ø–ª–µ—è—Ö

    display_array(rectangle)  # –í—ã–≤–µ—Å—Ç–∏ –ø—Ä—è–º–æ—É–≥–æ–ª—å–Ω–∏–∫
    sleep(1)

except KeyboardInterrupt:
    print("\n–ü—Ä–æ–≥—Ä–∞–º–º–∞ –∑–∞–≤–µ—Ä—à–µ–Ω–∞.")
