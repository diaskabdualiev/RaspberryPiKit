   from gpiozero import Button, Buzzer
   from signal import pause

   # –û–ø—Ä–µ–¥–µ–ª—è–µ–º –Ω–æ–º–µ—Ä–∞ GPIO –≤ —Ä–µ–∂–∏–º–µ BCM
   BUZZER_PIN = 18
   BUTTON_PIN = 23

   # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –±—É–∑–∑–µ—Ä–∞ –∏ –∫–Ω–æ–ø–∫–∏

   button = Button(BUTTON_PIN, pull_down=True)  # –ò—Å–ø–æ–ª—å–∑—É–µ–º –≤–Ω—É—Ç—Ä–µ–Ω–Ω—é—é –ø–æ–¥—Ç—è–∂–∫—É –∫ –Ω–∏–∑–∫–æ–º—É —É—Ä–æ–≤–Ω—é

   # –§—É–Ω–∫—Ü–∏–∏ –æ–±—Ä–∞–±–æ—Ç–∫–∏ —Å–æ–±—ã—Ç–∏–π
   def button_pressed():
       buzzer.on()
       print("–ö–Ω–æ–ø–∫–∞ –Ω–∞–∂–∞—Ç–∞! –í–∫–ª—é—á–∞–µ–º –±—É–∑–∑–µ—Ä.")

   def button_released():
       buzzer.off()
       print("–ö–Ω–æ–ø–∫–∞ –æ—Ç–ø—É—â–µ–Ω–∞! –í—ã–∫–ª—é—á–∞–µ–º –±—É–∑–∑–µ—Ä.")

   # –ü—Ä–∏–≤—è–∑—ã–≤–∞–µ–º –æ–±—Ä–∞–±–æ—Ç—á–∏–∫–∏ —Å–æ–±—ã—Ç–∏–π
   button.when_pressed = button_pressed
   button.when_released = button_released

   print("–û–∂–∏–¥–∞–Ω–∏–µ –Ω–∞–∂–∞—Ç–∏—è –∫–Ω–æ–ø–∫–∏ (–Ω–∞–∂–º–∏—Ç–µ Ctrl+C –¥–ª—è –≤—ã—Ö–æ–¥–∞)...")

   # –ó–∞–ø—É—Å–∫ –±–µ—Å–∫–æ–Ω–µ—á–Ω–æ–≥–æ —Ü–∏–∫–ª–∞ –æ–∂–∏–¥–∞–Ω–∏—è
   pause()
